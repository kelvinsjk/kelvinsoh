\begin{questions}

\question
	\textbf{Fractional Arithmetic}


		Evaluate the following:
	
	\begin{parts}
		\part[1]
			${- \frac{6}{7} + 2}$
		\droppoints

		\part[1]
			${- \frac{7}{9} + (- \frac{5}{9})}$
		\droppoints

		\part[1]
			${- \frac{5}{8} + 0}$
		\droppoints

		\part[1]
			${- \frac{5}{9} + (- 1)}$
		\droppoints

		\part[1]
			${2 + (- 4)}$
		\droppoints

	\end{parts}

\question
	\textbf{Linear Equations}


		Solve the following equations:
	
	\begin{parts}
		\part[2]
			${7 x - 4 = 5}$
		\droppoints

		\part[2]
			${- 5 x - 2 = -1}$
		\droppoints

		\part[2]
			${6 x + 3 = -5}$
		\droppoints

		\part[2]
			${5 x - 8 = -5}$
		\droppoints

		\part[2]
			${- 4 x + 7 = 7}$
		\droppoints

	\end{parts}


\end{questions}
