    ${(3 x^3)^2 - (2 x - 6)^2.}$
\droppoints


\question[2]

    ${x^2 - 4 x + 1}$
    in the form ${(x+h)^2 + k.}$
\droppoints


\question[2]
	Solve
    ${2 x^2 - 9 x - 10 < 8.}$
\droppoints


\question[3]
	Solve
    $$\frac{ 14 x^2 - 9 x - 18 }{ x - 4 } - (5 x + 9) = -5.$$
\droppoints


\question[2]
	Long divide
    ${\displaystyle \frac{x^2 + 9 x - 2}{x - 2}.}$
\droppoints


\question
	
	\begin{parts}
		\part[1]
			Make ${x}$ the subject from the following equation:
    $$\ln (7 - x) = - 5 t.$$
		\droppoints

		\part[1]
			What can you say about ${x}$ for large values of ${t?}$
		\droppoints

	\end{parts}

\question
	
	\begin{parts}
		\part[3]

		\part[1]
			*** What can you say about ${y}$ for large values of ${t?}$
		\droppoints

	\end{parts}

		\part[1]

\question[3]
	Solve, for ${0 \leq \theta \leq \frac{\pi}{2},}$
    $$\cos 2 \theta  - 5 \sin \theta  + 2 = 0.$$
\droppoints


\question[3]
	Find the values of the constants ${a}$ and
    ${b}$ such that
    $$8 \cos^2 \theta - 5 = a\sin^2 \theta + b$$
    for all values of ${\theta.}$
\droppoints
